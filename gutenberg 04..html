emma_raw = nltk.corpus.gutenberg.raw("austen-emma.txt")

from nltk.tokenize import RegexpTokenizer
retokenize = RegexpTokenizer("[\w]+")
retokenize.tokenize(emma_raw[50:100])

from nltk.tag import pos_tag
from nltk import FreqDist

emma_tokens = pos_tag(retokenize.tokenize(emma_raw))
n_list = [t[0] for t in emma_tokens if t[1] == "NN"] # 명사(NN)
fd_n = FreqDist(n_list)

emma_tokens = pos_tag(retokenize.tokenize(emma_raw))
v_list = [t[0] for t in emma_tokens if t[1] == "VB"] # 동사(VB)
fd_v = FreqDist(v_list)

emma_tokens = pos_tag(retokenize.tokenize(emma_raw))
a_list = [t[0] for t in emma_tokens if t[1] == "JJ"] # 형용사(JJ)
fd_a = FreqDist(a_list)
